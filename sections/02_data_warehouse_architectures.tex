\section{DWH Architectures}

% -------- subsection: Overview --------

\begin{frame}
  \frametitle{Approaches to Building a Decision-Support System}

  There is no single ``correct'' DWH architecture. The right choice depends
  on the organisation's size, budget, maturity, and data strategy.

  \begin{block}{Four Main Approaches}
    \begin{enumerate}
      \item \textbf{Virtual Data Warehouse} --- direct access to operational systems
      \item \textbf{Independent Data Marts} --- separate marts per department
      \item \textbf{Monolithic Enterprise DWH} --- single centralised warehouse
      \item \textbf{Integrated (Evolutionary) Architecture} --- EDW + dependent data marts
    \end{enumerate}
  \end{block}

  \begin{alertblock}{Key Insight}
    The evolution of most DWH landscapes follows this sequence.
    Organisations often start with option 1 or 2 and eventually move
    towards option 4.
  \end{alertblock}
\end{frame}

% -------- subsection: Virtual DWH --------

\begin{frame}
  \frametitle{Architecture 1: Virtual Data Warehouse}

  \begin{block}{Concept}
    Instead of physically copying data into a warehouse, queries are directed
    \textbf{directly to the operational source systems} --- or to extracts thereof.
    No central data store is created.
  \end{block}

  % PICTURE PLACEHOLDER
  \begin{alertblock}{Picture: Virtual DWH Architecture}

    Customers, Accounts, Invoices) on the left, connected via direct
    query/access arrows to BI tools (Finance, Marketing, Sales) on the right.
    Label it ``Virtual Warehouse''.}
  \end{alertblock}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Advantages}
      \begin{itemize}
        \item Low development effort and cost
        \item Fast results
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{alertblock}{Disadvantages}
      \begin{itemize}
        \item Uncertain data quality
        \item Burdens operational systems
        \item \textbf{No historical data available}
      \end{itemize}
    \end{alertblock}
  \end{columns}

  \vspace{0.3em}
  \textit{A virtual DWH is not really a warehouse.}
\end{frame}

% -------- subsection: Independent Data Marts --------

\begin{frame}
  \frametitle{Architecture 2: Independent Data Marts}

  \begin{block}{Concept}
    Each department builds its own data mart, sourced
    \textbf{directly from operational systems}.
    There is no central integration layer.
  \end{block}

  % PICTURE PLACEHOLDER
  \begin{alertblock}{Picture: Independent Data Marts}
    \textit{Add a diagram showing operational source systems feeding
    directly into separate Finance, Marketing, and Sales data marts.
    Show many-to-many extraction lines forming a ``spider web''.}
  \end{alertblock}

  \begin{block}{The m $\times$ n Extraction Problem}
    With \textbf{m} independent data marts and \textbf{n} source systems,
    up to $m \times n$ extraction interfaces are needed.
    This quickly becomes an unmanageable web of connections.
  \end{block}
\end{frame}

% ------------------------------------------------------------
\begin{frame}
  \frametitle{Independent Data Marts: Pros and Cons}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Advantages}
      \begin{itemize}
        \item Lower costs
        \item Shorter implementation time
        \item Fast ROI per department
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{alertblock}{Disadvantages}
      \begin{itemize}
        \item Fragmented, isolated data (``data silos'')
        \item Multiple conflicting ``versions of the truth''
        \item Redundant and inefficient extracts
        \item No consistent quality standards
        \item Expensive and redundant storage
        \item Maintenance becomes a nightmare
        \item Long-term costs outweigh short-term benefits
      \end{itemize}
    \end{alertblock}
  \end{columns}
\end{frame}

% -------- subsection: Monolithic DWH --------

\begin{frame}
  \frametitle{Architecture 3: Monolithic Enterprise Data Warehouse}

  \begin{block}{Concept}
    A single, large, centralised data warehouse that integrates all enterprise
    data. End users query the warehouse directly --- there are no separate
    data marts.
  \end{block}

  % PICTURE PLACEHOLDER
  \begin{alertblock}{Picture: Monolithic EDW}
    \textit{Add diagram: operational sources feeding into one large
    central Enterprise Data Warehouse, with BI tools querying it directly.
    No data marts layer.}
  \end{alertblock}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Advantage}
      Integrated and consistent data across all departments.
    \end{block}

    \column{0.48\textwidth}
    \begin{alertblock}{Disadvantages}
      \begin{itemize}
        \item Not suited for end-user queries (too complex/detailed)
        \item Conflicting requirements from different departments
        \item Poor query response times
        \item Long project duration before any business value
        \item High costs before ROI
      \end{itemize}
    \end{alertblock}
  \end{columns}
\end{frame}

% -------- subsection: Integrated / Hub-and-Spoke --------

\begin{frame}
  \frametitle{Architecture 4: Integrated (Hub-and-Spoke)}

  \begin{block}{Concept}
    The recommended architecture. The \textbf{Enterprise Data Warehouse (EDW)}
    acts as the central hub, and \textbf{dependent data marts} are built
    on top of it for departmental use.
    Built \textbf{iteratively}: start with one data mart, expand step by step.
  \end{block}

  % PICTURE PLACEHOLDER
  \begin{alertblock}{Picture: Hub-and-Spoke Architecture}
    \textit{Add diagram showing: Operational sources $\to$ EDW (Hub) $\to$
    Finance Mart, Marketing Mart, Sales Mart, Personal Data Marts (Spokes).
    Show metadata flowing horizontally. Include ODS as an optional component
    between sources and the EDW.}
  \end{alertblock}
\end{frame}

% ------------------------------------------------------------
\begin{frame}
  \frametitle{The Integrated Architecture: Building Iteratively}

  \begin{block}{How the Integrated Architecture Grows}
    \begin{enumerate}
      \item Start with an \textbf{enterprise-wide subject model} and
        infrastructure plan
      \item Build the \textbf{first data mart}, simultaneously creating
        the first layer of the EDW
      \item Build the \textbf{second data mart}, extending the EDW with
        shared/common data
      \item Continue adding marts and expanding the EDW incrementally
      \item Eventually add an \textbf{ODS} for near-real-time operational reporting
      \item Incorporate \textbf{external and unstructured data}
    \end{enumerate}
  \end{block}

  \begin{block}{"Think Big -- Start Small"}
    The iterative approach allows the organisation to learn, reduce risk,
    and deliver early value while building towards a comprehensive solution.
  \end{block}
\end{frame}

% ------------------------------------------------------------
\begin{frame}
  \frametitle{The Integrated Architecture: Pros and Cons}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{Advantages}
      \begin{itemize}
        \item Consistent, integrated data
        \item Single source of truth
        \item Flexible and extensible
        \item Reduced up-front investment
        \item Early ROI via first data marts
        \item Lower risk via iterative delivery
        \item Data marts are optimised for end-user queries
      \end{itemize}
    \end{block}

    \column{0.48\textwidth}
    \begin{alertblock}{Challenges}
      \begin{itemize}
        \item Requires enterprise-wide vision and governance
        \item Higher initial planning effort
        \item Long path to full coverage
        \item Needs experienced architects
      \end{itemize}
    \end{alertblock}
  \end{columns}
\end{frame}

% -------- subsection: The Enterprise Data Warehouse --------

\begin{frame}
  \frametitle{The Enterprise Data Warehouse (EDW)}

  \begin{block}{Definition and Purpose}
    The EDW was introduced by Bill Inmon around 1990.
    It \textbf{centralises, standardises, and integrates} data from all
    operational sources.
  \end{block}

  \begin{block}{Goals of the EDW}
    \begin{itemize}
      \item Build the \textbf{history} that is missing in OLTP systems
      \item Integrate data from across the entire enterprise
      \item Provide a \textbf{clean and reliable foundation} for the warehouse environment
      \item Feed departmental and personal data marts
      \item Be as \textbf{flexible as possible} for future extensions
    \end{itemize}
  \end{block}

  \begin{alertblock}{Important}
    The EDW is \textbf{not} intended for direct end-user queries.
    It is the backend engine that feeds the data marts.
  \end{alertblock}
\end{frame}

% -------- subsection: Data Mart --------

\begin{frame}
  \frametitle{The Data Mart}

  \begin{block}{Definition}
    A \textbf{Data Mart} is a subset of the DWH environment focused on the
    data and analytical needs of a \textbf{specific department or business domain}.
  \end{block}

  \begin{block}{Characteristics}
    \begin{itemize}
      \item Smaller in scope than the EDW
      \item Aligned with a specific business requirement (Sales, Finance, HR)
      \item Minimises complexity of data access for end users
      \item Delivers good query performance
      \item Uses dimensional models (star/snowflake schema)
    \end{itemize}
  \end{block}

  \begin{block}{Types}
    \begin{tabularx}{\linewidth}{@{}l X@{}}
      \textbf{Dependent}   & Fed from the EDW. Consistent with enterprise data. \\
      \textbf{Independent} & Sourced directly from operational systems. Risk of data silos. \\
      \textbf{Personal}    & Individual analyst subset. Loosely governed. \\
    \end{tabularx}
  \end{block}
\end{frame}

% -------- subsection: ODS --------

\begin{frame}
  \frametitle{The Operational Data Store (ODS)}

  \begin{block}{Definition (Inmon, early 1990s)}
    The ODS is a \textbf{subject-oriented, integrated, volatile, and current}
    data store that supports operational decision making.
  \end{block}

  \begin{block}{Key Properties}
    \begin{itemize}
      \item Contains \textbf{current} (or very recent) integrated data --- no full history
      \item Data is \textbf{updated frequently} (near real-time or hourly)
      \item Can \textbf{feed the DWH} with already-cleaned, integrated data
      \item Or can be fed \textit{by} the DWH for operational reporting
    \end{itemize}
  \end{block}

  \begin{block}{ODS vs.\ EDW}
    \begin{tabularx}{\linewidth}{@{}X X@{}}
      \textbf{ODS}                    & \textbf{EDW} \\
      Subject-oriented                & Subject-oriented \\
      Integrated                      & Integrated \\
      \textbf{Temporary / volatile}   & \textbf{Non-volatile} \\
      \textbf{Current data only}      & \textbf{Full historical data} \\
    \end{tabularx}
  \end{block}
\end{frame}

% -------- subsection: Metadata --------

\begin{frame}
  \frametitle{Metadata: The Data About Data}

  \begin{block}{Definition}
    \textbf{Metadata} is descriptive information about the structure and
    content of data, and about the applications and processes that access it.
    \hfill\small(MDC OIM Specification)
  \end{block}

  \begin{block}{Technical Metadata (maintained by IT)}
    \begin{itemize}
      \item Storage location, infrastructure (network, servers)
      \item Number of columns and rows in a table
      \item Table structure, format, relationships
      \item ETL rules: extraction, transformation, load processes
      \item Creation date and update schedules
    \end{itemize}
  \end{block}

  \begin{block}{Business Metadata (maintained by users)}
    \begin{itemize}
      \item Business rules, transformation logic, calculation descriptions
      \item Business-language descriptions of tables and fields
      \item Data ownership and security classifications
      \item Definition and purpose of each data element
    \end{itemize}
  \end{block}
\end{frame}
